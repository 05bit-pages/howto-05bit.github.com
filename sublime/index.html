How to ignore files and/or dirs in projects?

{
    "folders":
    [
        {
            "path": "src",
            "folder_exclude_patterns": ["backup"],
            "follow_symlinks": true
        },
        {
            "path": "docs",
            "file_exclude_patterns": ["*.tmp"]
        }
    ]
}
